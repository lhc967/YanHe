        // --- 1. Tab Logic ---
        function switchTab(tabId, btn) {
            document.querySelectorAll('.tab-pane').forEach(el => {
                el.className = 'tab-pane d-none';
            });
            document.getElementById('tab-' + tabId).className = 'tab-pane d-block animation-fade';

            document.querySelectorAll('.nav-link-course').forEach(el => el.classList.remove('active'));
            btn.classList.add('active');
        }

        // --- 2. AI Toggle ---
        function toggleAI(btn) {
            btn.classList.toggle('active');
            if(btn.classList.contains('active')){
                const originalHTML = btn.innerHTML;
                btn.innerHTML = '<span class="spinner-grow spinner-grow-sm text-light" style="width: 0.5rem; height: 0.5rem;"></span> Activating...';
                setTimeout(() => {
                    btn.innerHTML = originalHTML;
                }, 800);
            }
        }

        // --- 3. Enrollment (Free Flow) ---
        function startLearning() {
            const btn = document.getElementById('btnJoin');
            btn.innerText = 'Processing...';
            btn.disabled = true;

            setTimeout(() => {
                alert('ðŸŽ‰ Welcome Aboard!\nYou have successfully enrolled in this course.');
                btn.innerText = 'Continue Learning';
                btn.classList.add('enrolled');
                btn.disabled = false;
            }, 800);
        }

        // --- 4. Video Play Simulation ---
        function playVideo() {
            document.getElementById('videoCover').style.opacity = '0';
            document.getElementById('playOverlay').style.opacity = '0';
            setTimeout(() => {
                document.getElementById('videoCover').style.display = 'none';
                document.getElementById('playOverlay').style.display = 'none';
                document.getElementById('playingState').style.display = 'flex';
            }, 500);
        }

        // --- 5. Chapter Switch ---
        function playChapter(item, index) {
            document.querySelectorAll('.chapter-item').forEach(el => {
                el.classList.remove('active');
                el.querySelector('i').classList.remove('text-primary');
                el.querySelector('i').classList.add('text-muted');
                const badge = el.querySelector('.badge');
                if(badge) badge.remove();
            });

            item.classList.add('active');
            item.querySelector('i').classList.remove('text-muted');
            item.querySelector('i').classList.add('text-primary');

            const badge = document.createElement('span');
            badge.className = 'badge bg-success text-white';
            badge.innerText = 'Playing';
            item.appendChild(badge);

            playVideo();
        }

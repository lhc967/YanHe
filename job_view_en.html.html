    <script>
        function openApplyModal() {
            var myModal = new bootstrap.Modal(document.getElementById('applyModal'));
            myModal.show();
        }

        function submitSuccess() {
            // Close modal
            const modalEl = document.getElementById('applyModal');
            const modal = bootstrap.Modal.getInstance(modalEl);
            modal.hide();

            // Simulate loading state on button
            const btn = document.querySelector('.btn-apply');
            btn.innerHTML = '<span class="spinner-border spinner-border-sm"></span> Sending...';
            btn.classList.remove('btn-success');
            btn.classList.add('btn-secondary');

            setTimeout(() => {
                alert('✅ Proposal Submitted Successfully!\nYou will be notified if the client shortlists you.');
                btn.innerText = 'Proposal Sent';
                btn.disabled = true;
            }, 1000);
        }
    </script>

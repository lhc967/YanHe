                nextUrl: 'course_detail_unlocked.html' // 关键：支付后去哪里
            },
            'vip': {
                img: 'https://images.unsplash.com/photo-1596797038530-2c107229654b?w=200',
                title: '言禾 Pro 年度会员',
                desc: '全站课程免费看 | 无限次 AI 纠音 | 任务接单免佣金',
                priceHtml: `<div class="d-flex justify-content-between mb-2 text-secondary"><span>年费</span><span>¥ 298.00</span></div>`,
                total: '¥ 298.00',
                nextUrl: 'profile.html'
            },
            'task': {
                img: 'https://images.unsplash.com/photo-1454165804606-c3d57bc86b40?w=200',
                title: '托管赏金：法律文件翻译',
                desc: '资金托管至平台担保账户，验收满意后打款',
                priceHtml: `<div class="d-flex justify-content-between mb-2 text-secondary"><span>赏金</span><span>¥ 500.00</span></div>`,
                total: '¥ 500.00',
                nextUrl: 'service.html'
            }
        };

        // 默认加载逻辑 (支持 URL 参数 ?type=course)
        const urlParams = new URLSearchParams(window.location.search);
        const type = urlParams.get('type') || 'course'; // 默认是课程
        loadScenario(type);

        function loadScenario(key) {
            const data = scenarios[key];
            document.getElementById('pImg').src = data.img;
            document.getElementById('pTitle').innerText = data.title;
            document.getElementById('pDesc').innerText = data.desc;
            document.getElementById('priceDetails').innerHTML = data.priceHtml;
            document.getElementById('totalPrice').innerText = data.total;
            document.getElementById('btnPay').innerText = `支付 ${data.total}`;

            // 设置成功页的跳转目标
            document.getElementById('btnBackCourse').onclick = function() {
                window.location.href = data.nextUrl;
            };
        }

        function selectPay(el) {
            document.querySelectorAll('.payment-method').forEach(e => {
                e.classList.remove('active');
                e.querySelector('.bi-check-circle-fill')?.classList.replace('bi-check-circle-fill', 'bi-circle');
                e.querySelector('.bi-check-circle-fill')?.classList.add('text-muted');
            });
            el.classList.add('active');
            const icon = el.querySelector('.bi-circle, .bi-check-circle-fill');
            icon.className = 'bi bi-check-circle-fill text-primary fs-5';
        }

        function handlePay() {
            const btn = document.getElementById('btnPay');
            btn.innerHTML = '<span class="spinner-border spinner-border-sm"></span> 正在安全支付...';
            setTimeout(() => {
                document.getElementById('checkoutView').style.display = 'none';
                document.querySelector('.demo-bar').style.display = 'none';
                document.getElementById('successView').style.display = 'block';
            }, 1500);
        }
    </script>

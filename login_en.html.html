    <script>
        function toggleView() {
            const login = document.getElementById('loginView');
            const signup = document.getElementById('signupView');

            if (login.style.display === 'none') {
                login.style.display = 'block';
                signup.style.display = 'none';
            } else {
                login.style.display = 'none';
                signup.style.display = 'block';
            }
        }

        // Switch between Individual and Enterprise Login
        function switchLoginType(type, btn) {
            // UI Toggle
            document.querySelectorAll('.tab-btn').forEach(el => el.classList.remove('active'));
            btn.classList.add('active');

            // Logic Changes
            const label = document.getElementById('labelEmail');
            const input = document.getElementById('inputEmail');

            if (type === 'enterprise') {
                label.innerText = 'Work Email (SSO)';
                input.placeholder = 'name@company.com';
                // You could also hide "Forgot Password" for SSO logic here
            } else {
                label.innerText = 'Email';
                input.placeholder = 'name@example.com';
            }
        }

        function handleAuth(targetUrl) {
            const isLogin = document.getElementById('loginView').style.display !== 'none';
            const btn = isLogin ? document.getElementById('btnLogin') : document.getElementById('btnSignup');

            btn.innerHTML = '<span class="spinner-border spinner-border-sm"></span> Verifying...';
            btn.style.opacity = '0.7';
            btn.disabled = true;

            setTimeout(() => {
                window.location.href = targetUrl;
            }, 1000);
        }
    </script>

    <script>
        let currentRole = 'client';
        let isCompany = true;

        function switchRole(role) {
            currentRole = role;
            document.querySelectorAll('.role-card').forEach(el => el.classList.remove('active'));
            document.getElementById('role' + capitalize(role)).classList.add('active');

            if (role === 'client') {
                document.getElementById('formClient').classList.remove('d-none-custom');
                document.getElementById('formFreelancer').classList.add('d-none-custom');
            } else {
                document.getElementById('formClient').classList.add('d-none-custom');
                document.getElementById('formFreelancer').classList.remove('d-none-custom');
            }
        }

        function toggleCompanyFields(val) {
            isCompany = val;
            const field = document.getElementById('companyFields');
            const tip = document.getElementById('budgetTip');

            if (isCompany) {
                field.style.display = 'block';
                tip.innerText = 'Corporate payment (Invoice available).';
            } else {
                field.style.display = 'none';
                tip.innerText = 'Funds secured in Escrow. Released upon approval.';
            }
        }

        function aiGenerateDesc() {
            const btn = document.querySelector('.ai-btn');
            const textarea = document.getElementById('taskDesc');
            btn.innerHTML = '<span class="spinner-border spinner-border-sm"></span> Generating...';

            setTimeout(() => {
                const text = isCompany
                    ? "[Project Scope]\nWe are looking for a certified interpreter for an upcoming business delegation from Germany.\n\n[Responsibilities]\n- Consecutive interpretation during meetings.\n- Drafting meeting minutes.\n\n[Requirements]\n- Native Chinese, fluent in German.\n- 3+ years experience in the Automotive sector."
                    : "[Personal Request]\nI need someone to translate a 500-word abstract for my Architecture thesis.\n\n[Requirements]\n- Must be familiar with academic terminology.\n- Need it delivered within 48 hours.";

                textarea.value = text;
                btn.innerHTML = '<i class="bi bi-check"></i> Generated';
                btn.classList.add('btn-success');

                // Auto fill budget
                document.getElementById('budgetInput').value = isCompany ? 500 : 50;
            }, 800);
        }

        function handleSubmit() {
            const btn = document.querySelector('.btn-next');
            btn.innerText = 'Processing...';
            setTimeout(() => {
                if(currentRole === 'client') {
                    // Redirect to Checkout
                    window.location.href = 'checkout_en.html';
                } else {
                    alert('ðŸŽ‰ Service Posted! Pending review.');
                    window.location.href = 'service_en.html';
                }
            }, 1000);
        }

        function capitalize(s) { return s.charAt(0).toUpperCase() + s.slice(1); }
    </script>

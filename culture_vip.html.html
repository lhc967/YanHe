    <script>
        var myChart = echarts.init(document.getElementById('china-map'));
        // 普通地图样式
        var stdOption = {
            geo: { map: 'china', roam: true, zoom: 1.2, label: {show:false}, itemStyle: { areaColor: '#e0eaff', borderColor: '#fff' } },
            series: [{ type: 'effectScatter', coordinateSystem: 'geo', data: [{name:'成都',value:[104.06,30.67]}], symbolSize: 15, itemStyle: { color: '#004aad' } }]
        };
        // Pro 黑金地图样式
        var proOption = {
            geo: { map: 'china', roam: true, zoom: 1.2, label: {show:false}, itemStyle: { areaColor: '#2a2a2a', borderColor: '#444' } },
            series: [{ type: 'effectScatter', coordinateSystem: 'geo', data: [{name:'成都',value:[104.06,30.67]}], symbolSize: 20, itemStyle: { color: '#ffd700' }, rippleEffect: { color: '#ffd700' } }]
        };

        myChart.setOption(stdOption);
        myChart.on('click', function() { document.getElementById('drawer').classList.add('show'); });

        function toggleMode(mode) {
            if(mode === 'pro') {
                document.getElementById('btnStd').classList.remove('active');
                document.getElementById('btnPro').classList.add('pro-active');
                document.body.classList.add('pro-mode');
                myChart.setOption(proOption);
                // 自动切换到挑战Tab
                switchTab('scenario', document.getElementById('tabLinkScenario'));
                document.getElementById('drawer').classList.add('show');
            } else {
                document.getElementById('btnStd').classList.add('active');
                document.getElementById('btnPro').classList.remove('pro-active');
                document.body.classList.remove('pro-mode');
                myChart.setOption(stdOption);
            }
        }

        function switchTab(id, btn) {
            document.querySelectorAll('.tab-pane').forEach(el => el.classList.remove('active'));
            document.getElementById('tab-'+id).classList.add('active');
            if(btn) {
                document.querySelectorAll('.tab-link').forEach(el => {
                    el.classList.remove('active');
                    el.classList.remove('pro-active');
                });
                btn.classList.add(id === 'scenario' ? 'pro-active' : 'active');
            }
        }
    </script>

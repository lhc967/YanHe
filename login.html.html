    <script>
        // 当前状态
        let currentRole = 'enterprise'; // 'enterprise' or 'individual'
        let currentView = 'login';      // 'login' or 'register'

        // 1. 切换角色 (企业 vs 个人)
        function switchRole(role) {
            currentRole = role;

            // UI 更新：Tab 样式
            document.querySelectorAll('.role-item').forEach(el => el.classList.remove('active', 'individual'));
            document.getElementById('tab' + capitalize(role)).classList.add('active');
            if(role === 'individual') document.getElementById('tabIndividual').classList.add('individual');

            // UI 更新：左侧品牌区
            const brandSec = document.getElementById('brandSection');
            const txtEnt = document.getElementById('brandTextEnterprise');
            const txtInd = document.getElementById('brandTextIndividual');

            if (role === 'individual') {
                brandSec.classList.add('individual-mode');
                txtEnt.classList.add('view-hidden');
                txtInd.classList.remove('view-hidden');
            } else {
                brandSec.classList.remove('individual-mode');
                txtEnt.classList.remove('view-hidden');
                txtInd.classList.add('view-hidden');
            }

            // UI 更新：表单字段与颜色
            updateFormVisibility();
        }

        // 2. 切换视图 (登录 vs 注册)
        function toggleView(view) {
            currentView = view;

            // 简单的显隐切换
            if(view === 'register') {
                document.getElementById('viewLogin').classList.add('view-hidden');
                document.getElementById('viewRegister').classList.remove('view-hidden');
            } else {
                document.getElementById('viewLogin').classList.remove('view-hidden');
                document.getElementById('viewRegister').classList.add('view-hidden');
            }
            updateFormVisibility();
        }

        // 3. 更新表单字段显隐 (核心逻辑)
        function updateFormVisibility() {
            // 通用变量
            const isIndividual = currentRole === 'individual';
            const themeColor = isIndividual ? '#ff6b00' : '#004aad';
            const themeBtnClass = isIndividual ? 'individual-btn' : '';
            const themeLinkClass = isIndividual ? 'individual-link' : '';

            // 按钮与链接颜色
            document.querySelectorAll('.btn-submit').forEach(btn => {
                btn.className = 'btn-submit ' + themeBtnClass; // 重置类名
            });
            document.querySelectorAll('.switch-link').forEach(link => {
                link.className = 'switch-link ' + themeLinkClass;
            });

            // 输入框焦点颜色 (通过类名控制 CSS)
            document.querySelectorAll('.form-control').forEach(input => {
                if(isIndividual) input.classList.add('individual-focus');
                else input.classList.remove('individual-focus');
            });

            // 字段显隐
            if (currentView === 'login') {
                document.getElementById('loginSubtitle').innerText = isIndividual ? '高校师生 / 个人创作者登录' : '使用企业认证账号登录工作台';

                document.getElementById('fieldsLoginEnterprise').classList.toggle('view-hidden', isIndividual);
                document.getElementById('fieldsLoginIndividual').classList.toggle('view-hidden', !isIndividual);
            } else {
                document.getElementById('regSubtitle').innerText = isIndividual ? '加入言禾人才库，开启兼职之旅' : '提交企业资质，发布语言需求';

                document.getElementById('fieldsRegEnterprise').classList.toggle('view-hidden', isIndividual);
                document.getElementById('fieldsRegIndividual').classList.toggle('view-hidden', !isIndividual);
            }
        }

        // 模拟登录操作
        function handleLogin() {
            const btn = document.getElementById('btnLogin');
            const originalText = btn.innerText;
            btn.innerText = '验证中...';
            btn.disabled = true;

            setTimeout(() => {
                if(currentRole === 'individual') {
                    // 个人 -> 去故事页或首页
                    window.location.href = 'story.html';
                } else {
                    // 企业 -> 去任务大厅发布任务
                    window.location.href = 'service.html';
                }
            }, 1000);
        }

        // 模拟注册操作
        function handleRegister() {
            alert('注册申请已提交！\n我们会发送验证邮件到您的邮箱。');
            toggleView('login');
        }

        // 辅助函数
        function capitalize(s) { return s.charAt(0).toUpperCase() + s.slice(1); }

    </script>
